43void Zappy::TileInfo::manageHiding()
44{
45 if (this->_gui->getTrantorianInfo()->isHiding())
46 return;
47 if (this->_pos.x <= this->_unselectedPos.x) {
48 this->_pos.x = this->_unselectedPos.x;
49 this->_tile = this->_nextTile;
50 this->updateTrantorButtonsTab();
51 this->_gui->getTrantorianInfo()->hide();
52 this->_nextTile = nullptr;
53 this->_hiding = false;
54 return;
55 }
56 this->_pos.x -= this->_animationSpeed;
57}

60{

100void Zappy::TileInfo::updateTrantorButtonsTab()
101{
102 if (!this->_tile)
103 return;
104 this->_buttons.clear();
105 size_t index = 0;
106
107 for (auto trantorian: this->_gui->getMap()->getAllTrantorians()) {
108 if (!trantorian.second)
109 continue;
110 if (trantorian.second->getTilePos().x != this->_tile->getIndex().x)
111 continue;
112 if (trantorian.second->getTilePos().y != this->_tile->getIndex().y)
113 continue;
114 std::shared_ptr<TrantorButtons> button = std::make_shared<TrantorButtons>(this->_gui, trantorian.second);
115 button->setSize(sf::Vector2f(280, 80));
116 button->setBgFillColor(sf::Color(230, 230, 230));
117 button->setTextFillColor(sf::Color::Black);
118 button->setTextOutlineColor(sf::Color::Black);
119 button->setMaxScale(sf::Vector2f(1.05, 1.05));
120 button->setMinScale(sf::Vector2f(0.95, 0.95));
121 button->setTitle("Trantorian #" + std::to_string(trantorian.second->getID()));
122 button->setPosition({160, static_cast<float>(500 + 100 * index)});
123 button->setBasePosition({160, static_cast<float>(500 + 100 * index)});
124 this->_buttons.push_back(button);
125 index++;
126 }
