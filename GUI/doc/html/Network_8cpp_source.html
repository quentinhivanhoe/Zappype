33std::string Zappy::Network::receive(bool isBlocking)
34{
35 std::array<char, 4096> recievedArray;
36 size_t nb_bytes;
37 this->_socket.setBlocking(isBlocking);
38 if (_socket.receive(recievedArray.data(), recievedArray.size(), nb_bytes) == sf::Socket::Done) {
39 recievedArray[nb_bytes] = '\0';
40 return std::string(recievedArray.data());
41 }
42 return "";
43}

66void Zappy::Network::recieveFromServer()
67{
68 Parser parser;
69 sf::Clock clock;
70 while (!this->_isShuttingDown) {
71 if (clock.getElapsedTime().asSeconds() > 10) {
72 clock.restart();
73 this->send("mct\n");
74 }
75 std::string recievedString = this->receive(false);
76 if (!recievedString.empty()) {
77 std::vector<std::string> tab = Parser::splitLine(recievedString, "\n");
78 for (size_t i = 0; i < tab.size(); i++) {
79 std::vector<std::string> args = Parser::parseLine(tab[i], ' ');
80 parser.manageResponse(args, this);
81 }
82 }
83 }
84 this->_socket.disconnect();
85}

141void Zappy::Network::askPlayersInfo()
142{
143 if (this->_playerNb <= 0) {
144 std::cout << "Must have player number set before asking for players info." << std::endl;
145 return;
146 }
147 this->send("pls\n");
148 Parser parser;
149 std::string playersInfo = "";
150 while (true) {
151 std::string recievedString = this->receive(false);
152 playersInfo.append(recievedString);
153 if (static_cast<int>(Parser::nbOccur(playersInfo, "pls")) >= this->_playerNb)
154 break;
155 }
156 std::vector<std::string> tab = Parser::splitLine(playersInfo, "\n");
