184void Zappy::GUI::connect()
185{
186 this->_ipSelected = false;
187 this->_portSelected = false;
188 this->_hasConnectionFailed = false;
189 if (this->_tmpIp.empty() || this->_tmpPort.empty())
190 return;
191 try {
192 this->_networkInfo = std::make_shared<Network>(this);
193 this->_networkInfo->establishConnection(this->_tmpIp, std::stoi(this->_tmpPort));
194 this->runGame();
195 this->_isOnConnectionPage = false;
196 this->_networkInfo.reset();
197 } catch(const Error& e) {
198 std::cerr << "Zappy Exception - [From : " << e.where() << "] - [" << e.what() << "]" << std::endl;
199 this->_hasConnectionFailed = true;
200 this->_isOnConnectionPage = true;
201 this->_networkInfo.reset();
202 }
203}

312void Zappy::GUI::updateClock()
313{
314 if (this->time < 0.0f) this->time = 0.0f;
315 if (this->time >= 24.0f) this->time = 23.9999f;
316 int hours = static_cast<int>(this->time);
317 float fractionalHours = this->time - hours;
318 int minutes = static_cast<int>(fractionalHours * 60);
319 float fractionalMinutes = fractionalHours * 60 - minutes;
320 int seconds = static_cast<int>(std::round(fractionalMinutes * 60));
321
322 if (seconds == 60) {
323 seconds = 0;
324 minutes++;
325 }
326 if (minutes == 60) {
327 minutes = 0;
328 hours = (hours + 1) % 24;
329 }
330 std::ostringstream oss;
331 oss << std::setfill('0') << std::setw(2) << hours << ":"
332 << std::setfill('0') << std::setw(2) << minutes << ":"
333 << std::setfill('0') << std::setw(2) << seconds;
334
335 this->_TimeText.setString(oss.str());
336}

347void Zappy::GUI::updateSky()
348{
349 sf::Color dawnColor(150, 180, 255);
350 sf::Color noonColor(135, 206, 235);
351 sf::Color duskColor(255, 100, 50);
352 sf::Color nightColor(10, 10, 30);
353 sf::Color dawnBottom(100, 100, 180);
354 sf::Color noonBottom(200, 220, 255);
355 sf::Color duskBottom(255, 150, 80);
356 sf::Color nightBottom(10, 20, 40);
357 sf::Color Bottom;
358 sf::Color Top;
359
360 float timeElapsed = this->_clock.getElapsedTime().asSeconds();
361 this->time = timeElapsed * 0.1f;
362 if (this->time > 24.f) {
363 this->time -= 24;
364 this->_clock.restart();
365 }
366 if (time < 6.f) {
367 Top = lerpColor(nightColor, dawnColor, time / 6.f);
368 Bottom = lerpColor(nightBottom, dawnBottom, time / 6.f);
369 } else if (time < 12.f) {
370 this->night = false;
371 Top = lerpColor(dawnColor, noonColor, (time - 6.f) / 6.f);
372 Bottom = lerpColor(dawnBottom, noonBottom, (time - 6.f) / 6.f);
373 } else if (time < 18.f) {
374 Top = lerpColor(noonColor, duskColor, (time - 12.f) / 6.f);
375 Bottom = lerpColor(noonColor, duskBottom, (time - 12.f)/ 6.f);
376 } else if (time < 20.f) {
377 Top = lerpColor(duskColor, nightColor, (time - 18.f) / 2.f);
378 Bottom = lerpColor(duskBottom, nightBottom, (time - 18.f) / 2.f);
379 } else {
380 Top = lerpColor(nightColor, dawnColor, (time - 20.f) / 4.f);
381 Bottom = lerpColor(nightBottom, dawnBottom, (time - 20.F) / 4.f);
382 this->night = true;
383 }
384 background[0].color = Top;
385 background[1].color = Top;
386 background[2].color = Bottom;
387 background[3].color = Bottom;
388}

453void Zappy::GUI::handleMenuEvents()
454{
455 sf::Event event;
456 while (this->_window.pollEvent(event)) {
457 if (event.type == sf::Event::Closed)
458 this->_window.close();
459 if (event.type == sf::Event::TextEntered) {
460 char letter = event.text.unicode;
461 if (this->_ipSelected) {
462 if (letter == '\b' && !this->_tmpIp.empty())
463 this->_tmpIp.pop_back();
464 if (Parser::isNum({letter}) || letter == '.'){
465 if (this->_tmpIp.size() >= 15)
466 this->_tmpIp.pop_back();
467 this->_tmpIp += {letter};
468 }
469 }
470 if (this->_portSelected) {
471 if (letter == '\b' && !this->_tmpPort.empty())
472 this->_tmpPort.pop_back();
473 if (Parser::isNum({letter})){
474 if (this->_tmpPort.size() >= 4)
475 this->_tmpPort.pop_back();
476 this->_tmpPort += {letter};
477 }
478 }
479 }
480 }
481}

553sf::Vector2i Zappy::GUI::getCenterTilePos(float px, float py, float originX, float originY, float tileWidth, float tileHeight)
554{
555 int rectIndexX = (originX / tileWidth) - (px / tileWidth);
556 int rectIndexY = (py / tileHeight) - (originY / tileHeight);
557 sf::Vector2i finalIndex(0, 0);
558 finalIndex += sf::Vector2i(1, 1) * rectIndexY;
559 finalIndex.x += rectIndexX;
560 finalIndex.y -= rectIndexX;
561 return finalIndex;
562}

564void Zappy::GUI::update()
565{
566 while (this->_window.isOpen() && this->_gameIsRunning) {
567 if (this->_framerateClock.getElapsedTime().asMicroseconds() < 1000000 / 60)
568 continue;
569 if (!this->_window.isOpen())
570 break;
571 this->_framerateClock.restart();
572 this->_mouse.update(this->_window);
573 this->_uiMouse.update(this->_window);
574 if (this->_isEnding)
575 continue;
576 this->_tileInfo->update(this->_uiMouse);
577 this->_trantorianInfo->update(this->_uiMouse);
578 this->_broadcastTab->update();
579 this->updateTrantor();
580 this->handleTileButtons();
581 this->updateSky();
582 touchView();
583 }
584}

603void Zappy::GUI::run()
604{
605 this->_gameIsRunning = true;
606 this->_isEnding = false;
607 std::thread updateThread = std::thread(&GUI::update, this);
608 while (this->_window.isOpen() && this->_gameIsRunning) {
609 this->handleWindowEvents();
610 if (this->_isEnding) {
611 this->manageEndOfGame();
612 continue;
613 }
614 this->_window.clear(sf::Color::Black);
615 this->_window.setView(this->_window.getDefaultView());
616 this->_window.draw(this->background);
