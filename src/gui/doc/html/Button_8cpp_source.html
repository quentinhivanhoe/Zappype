201void Zappy::Buttons::update(MouseStatus mouse)
202{
203 float colorRed = this->_fillColor.r + 50 - 150 * mouse.hold();
204 float colorGreen = this->_fillColor.g + 50 - 150 * mouse.hold();
205 float colorBlue = this->_fillColor.b + 50 - 150 * mouse.hold();
206 float alpha = this->_fillColor.a;
207 Math::Clamp::function(colorRed, 0, 255);
208 Math::Clamp::function(colorGreen, 0, 255);
209 Math::Clamp::function(colorBlue, 0, 255);
210 this->_bg.setFillColor(this->_fillColor);
211 if (mouse.hover(this->_bg)) {
212 this->scaleUp(0.02);
213 this->_bg.setFillColor(sf::Color(colorRed, colorGreen, colorBlue, alpha));
214 if (mouse.pressed() && this->_function && this->_gui)
215 this->function();
216 } else {
217 this->scaleDown(0.02);
218 }
219}
